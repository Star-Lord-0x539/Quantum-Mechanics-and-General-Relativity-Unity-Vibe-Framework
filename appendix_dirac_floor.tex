\section{The seed-fixed spacetime-smeared Dirac floor}
\label{app:dirac-seed-floor}

This appendix records the derivation, from in-print worldline inputs,
of the one-sided absolute lower bound on the smeared renormalised
Dirac energy density at the audited-minimal strength consumed by the
(E2b) discussion: a \emph{single} fixed smearing triple, Hadamard
states of the seed folium, fixed seed metric. The organising fact is
that the Fewster--Verch route~\cite{FewsterVerch2001} never
represents the energy density as a positive quadratic form --- the
positivity it consumes is CAR-\emph{state} positivity of the
frequency-split two-point pieces (the Pauli window $0\le Q\le 1$),
with the negative-frequency (sea) piece entering through a
compensating minus sign --- so the indefinite-density obstruction
that blocks the Sanders positive-type-kernel organisation for Dirac
does not touch it.

\medskip\noindent\textbf{Standing data.} Seed
$(M,g_0)=(\mathbb R\times\Sigma,\,-dt^{2}\oplus h_\Sigma)$ with
$\Sigma$ a compact spin Cauchy slice, $\dim M=4$; Dirac mass $m>0$
(for $m=0$ see Rem.~\ref{rem:dsf-massless}); $\omega_0$ the
ultrastatic ground state with its canonical positive-frequency split
$P_0$ (the positive spectral projection of the time flow; the ground
state is Hadamard by Sahlmann--Verch~\cite{SahlmannVerch2000});
$h$ the one-particle Dirac Hamiltonian on $L^{2}(\Sigma;\mathbb C^4)$:
self-adjoint, discrete spectrum $\{E_k\}$ with $|E_k|\ge m$, smooth
eigensections $u_k$; $\gamma_x(t)=(t,x)$ the unit-speed static
worldlines. $\widehat T^{\mathrm{ren}}_{\mu\nu}$ is the fixed-$g_0$
Hadamard-renormalised Dirac stress
tensor~\cite{DappiaggiHackPinamonti2009}.

\begin{theorem}[seed-fixed spacetime-smeared absolute Dirac floor]
\label{thm:dirac-seed-floor}
Let $t^\mu=(\partial_t)^\mu$ and $F\in C_c^\infty(M;\mathbb R)$. Then
for every Hadamard state $\omega$ of the Dirac field on $(M,g_0)$,
\[
\big\langle\widehat T^{\mathrm{ren}}_{\mu\nu}[t^\mu t^\nu F^{2}]
\big\rangle_\omega \;\ge\; -c_{\mathrm{fl}}(g_0,\partial_t,F),\qquad
c_{\mathrm{fl}}:=\int_\Sigma\!d\mathrm{vol}(x)\,B_x
+\Big|\int_M F^{2}\,\big\langle\widehat T^{\mathrm{ren}}
(\partial_t,\partial_t)\big\rangle_{\omega_0}\Big|\;<\;\infty,
\]
where, with $g_x:=F(\cdot,x)\in C_0^\infty(\mathbb R;\mathbb R)$,
\[
B_x=\frac1{2\pi}\int_0^\infty\!d\mu\,\mu\,
\big(S_\mu(x)+S^\Gamma_\mu(x)\big),\qquad
S_\mu(x)=\sum_{k:\,E_k>0} w_k(x)\,|\widehat{g_x}(\mu+E_k)|^{2},
\]
and $S^\Gamma_\mu(x)=\sum_{k:\,E_k<0}w^\Gamma_k(x)\,
|\widehat{g_x}(\mu+|E_k|)|^{2}$ (the mirrored argument, from the
$(\mu,-\mu)$ evaluation of the Dawson--Fewster
bound~\cite{DawsonFewster2006}), with $w_k(x),w^\Gamma_k(x)\ge0$ the
spin-frame contractions of the mode densities of $u_k$ at $x$. In
particular the floor holds for the audited single triple, one-sided,
on Hadamard states of the seed folium.
\end{theorem}

\begin{proof}
(1) \emph{Disintegration.} For Hadamard $\omega$ the function
$x\mapsto\langle\widehat T^{\mathrm{ren}}_{\mu\nu}\rangle_\omega(x)$
is smooth (fixed-$g_0$ renormalisation,
\cite{DappiaggiHackPinamonti2009});
$d\mathrm{vol}_{g_0}=dt\,d\mathrm{vol}_\Sigma$ and $F$ is compactly
supported, so by Fubini
$\langle\widehat T^{\mathrm{ren}}[ttF^{2}]\rangle_\omega
=\int_\Sigma d\mathrm{vol}(x)\int_{\mathbb R}dt\,g_x(t)^{2}\,
\langle\widehat T^{\mathrm{ren}}(\partial_t,\partial_t)
\rangle_\omega(t,x)$.

(2) \emph{Reference conversion.} Pointwise on $M$,
$\langle\widehat T^{\mathrm{ren}}\rangle_\omega
=\langle{:}T{:}\rangle_\omega
+\langle\widehat T^{\mathrm{ren}}\rangle_{\omega_0}$, with
${:}T{:}$ the $\omega_0$-normal-ordered density: both subtractions
are Hadamard subtractions of the same classical density, so they
differ by the smooth $\omega$-independent $c$-number
$\langle\widehat T^{\mathrm{ren}}\rangle_{\omega_0}$ (the
renormalised, reference-free reading of the Fewster--Verch bound run
in the absolute direction). The $c$-number term integrates to a
finite constant on the compact support of $F^{2}$.

(3) \emph{Per-curve bound.} Fix $x\in\Sigma$. The static curve
$\gamma_x$ is smooth, timelike, unit-speed, and the weight $g_x^{2}$
with $g_x$ real $C_0^\infty$ is exactly the Dawson--Fewster class;
their explicit worldline quantum weak energy
inequality~\cite{DawsonFewster2006} with reference $\omega_0$ gives
$\int dt\,g_x(t)^{2}\,\langle{:}T{:}(\partial_t,\partial_t)
\rangle_\omega(t,x)\ge-B_x$ for every Hadamard $\omega$, with $B_x$
as displayed.

(4) \emph{Mode evaluation.} On the seed, the ground-state two-point
function restricted to positive frequency along $\gamma_x$ is the
mode sum $\sum_{E_k>0}w_k(x)\,e^{-iE_k(t-t')}$ (convergence in
$\mathscr D'(\mathbb R^{2})$, locally uniformly in $x$; the standard
ground-state expansion from $P_0$), and
$[\bar g_x\otimes g_x\,e^{-iE(t-t')}]^{\wedge}(-\mu,\mu)
=|\widehat{g_x}(\mu+E)|^{2}\ge0$; termwise summation is justified by
step (5). The $\Gamma$-piece runs identically on the negative
spectral subspace, the $(\mu,-\mu)$ evaluation producing the
mirrored argument $|\widehat{g_x}(\mu+|E_k|)|^{2}$.

(5) \emph{Envelope.} Elliptic sup-norm bound on compact $\Sigma$:
$\sup_x w_k(x)\le C(1+|E_k|)^{4}$ (Sobolev embedding
$H^{2}(\Sigma)\hookrightarrow C^{0}$ applied to $u_k$, with
$\|u_k\|_{H^{2}}\le C(1+E_k^{2})\|u_k\|_{2}$); Weyl asymptotics for
the first-order elliptic $h$: $\#\{k:|E_k|\le E\}\le C(1+E)^{3}$;
Paley--Wiener decay $\sup_x|\widehat{g_x}(\xi)|\le
C_N(1+|\xi|)^{-N}$ for all $N$. Hence
\[
\int_0^\infty\!d\mu\,\mu\,S_\mu(x)\le
C_N\sum_k(1+|E_k|)^{4}\int_0^\infty
\frac{\mu\,d\mu}{(1+\mu+|E_k|)^{N}}
\le C\sum_k(1+|E_k|)^{6-N}<\infty\qquad(N>9),
\]
uniformly in $x$; dominated convergence gives $x$-continuity of
$B_x$, hence $\int_\Sigma B_x\,d\mathrm{vol}<\infty$ ($\Sigma$
compact). All constants depend on $(g_0,F)$ only.

(6) \emph{Summation.} Integrating (3) over
$(\Sigma,d\mathrm{vol})$ and combining with (1)--(2) gives
$\langle\widehat T^{\mathrm{ren}}[ttF^{2}]\rangle_\omega
\ge-\int_\Sigma B_x\,d\mathrm{vol}
-\big|\int F^{2}\langle\widehat T^{\mathrm{ren}}
\rangle_{\omega_0}\big|=-c_{\mathrm{fl}}$.
\end{proof}

\paragraph{Spinor data (fixed once, background).} Over the chart-compact
$K''$ (frame bundle trivial there) fix a $g_0$-orthonormal oriented coframe
$e_0\in C^\infty$, $g_0=e_0^{\mathsf T}\eta e_0$, $\eta=\mathrm{diag}(-1,1,1,1)$,
with $E_0:=1+\|e_0\|_{C^{6,\alpha}}+\|e_0^{-1}\|_{C^{6,\alpha}}<\infty$
($U$-independent). For $g\in U$ set the frame-relative metric
$G[g]:=(e_0^{-1})^{\mathsf T}g\,e_0^{-1}$, so $G[g_0]=\eta$ and
$G[g]-\eta=(e_0^{-1})^{\mathsf T}(g-g_0)e_0^{-1}$. $S$ denotes the (rank
$N_{\mathrm{sp}}=4$) Dirac bundle, $\gamma^a$ the Clifford generators
($\gamma^a\gamma^b+\gamma^b\gamma^a=2\eta^{ab}$).

\begin{lemma}[spinor-frame ball-uniformity]\label{dspin:lem-frame}
Let $\Phi(M)=M^{\mathsf T}\eta M$. There is $\rho_\eta>0$ and a real-analytic
$\eta$-symmetric square root $\Psi:B_{\rho_\eta}(\eta)\to\{M:\eta M\ \mathrm{sym}\}$
with $\Phi\circ\Psi=\mathrm{id}$, $\Psi(\eta)=I$, and $L_k:=\sup_{\|G-\eta\|
\le\rho_\eta/2}\|D^k\Psi\|<\infty$ (real-analytic inverse function theorem;
only nondegeneracy of $\eta$ and $G\approx\eta$ are used, so the indefinite
signature is immaterial and no branch is crossed). Tighten $\varepsilon_0$ by
the one further $C^0$-smallness \textup{(u2$'$)} $\sup_{g\in U}\|G[g]-\eta\|
_{C^0(K'')}<\rho_\eta/2$ --- the same \emph{kind} of $C^0$-tightening as
\textup{(u2)}, the sole analytic cost of the frame. Define, for $g\in U$,
\[
  e[g]:=\Psi(G[g])\,e_0,\qquad e^{-1}[g]:=e_0^{-1}\Psi(G[g])^{-1},\qquad
  \omega[g]_\mu:=\tfrac14\bigl[e^{-1}(\partial_\mu e)+e^{-1}\Gamma[g]_\mu e\bigr]
  ^{ab}\gamma_a\gamma_b,
\]
so $e[g]^{\mathsf T}\eta e[g]=g$ and $|\det e[g]|=\sqrt{|\det g|}\ge
\sqrt{d_U}$. Then $e[g],e^{-1}[g],\omega[g]$ are single-valued real-analytic
functions of the $1$-jet of $g$, the $g_0$-spin structure transports uniquely
along the convex (hence contractible) ball $U$ --- no spin-structure jump ---
and there are ball-uniform $\Lambda_e,\Lambda_\omega,\kappa_e,\kappa_\omega
<\infty$, depending on $U$ only through $(g_0,\varepsilon_0,s,K'')$, with
\begin{align}
  \|e^{\pm1}[g]\|_{C^{6,\alpha}(K'')}&\le\Lambda_e\ \ (H^s\text{ register}),
  &\|\omega[g]\|_{C^{5,\alpha}(K'')}&\le\Lambda_\omega\ \ (H^{s-1}\text{ register}),
  \label{dspin:eq-framesup}\\
  \|e[g]-e[g']\|_{C^{6,\alpha}}&\le\kappa_e\,\|h\|_{C^{6,\alpha}},
  &\|\omega[g]-\omega[g']\|_{C^{5,\alpha}}&\le\kappa_\omega\,\|h\|_{C^{6,\alpha}},
  \label{dspin:eq-framediff}
\end{align}
$h:=g-g'$; no step differentiates in $g$ (real-side, pair-Lipschitz). The
displayed constants are
$\Lambda_e=cE_0(1+E_0^2\Lambda_U)^6\max_{k\le6}L_k$,
$\Lambda_\omega=c\Lambda_e^2(\Lambda_e+\Lambda_U^9 d_U^{-6})$,
$\kappa_e=cE_0^2(1+E_0^2\Lambda_U)^6\max_{k\le7}L_k$, and
$\kappa_\omega=c[\Lambda_e^2\kappa_e(\Lambda_e+\Lambda_U^9d_U^{-6})
+\Lambda_e(\kappa_e+\kappa_\Gamma(5)+\Lambda_U^9d_U^{-6}\kappa_e)]$ with
$\kappa_\Gamma(5)=c_5(n)\Lambda_U^{17}d_U^{-12}$ \emph{the scalar port's
Christoffel-difference constant, consumed verbatim}.
\end{lemma}
\begin{proof}
$e,e^{-1}$ are compositions/quotients of the fixed real-analytic $\Psi$ with
the affine $g\mapsto G[g]$ (image in $B_{\rho_\eta/2}(\eta)$ by (u2$'$)),
single-valued and real-analytic; $\omega$ adds $\partial e$ and $\Gamma[g]$,
real-analytic in the $1$-jet off $\{|\det g|\ge d_U\}$. Spin structure: the
reduction lives on the metric-independent $GL$-frame bundle, and over the
connected ball the $SO^\uparrow(1,3)$-subbundles are homotopic, so the
$H^1(\MM;\mathbb Z_2)$-torsor is rigid. \eqref{dspin:eq-framesup}:
$\|G[g]\|_{C^{6,\alpha}}\le E_0^2\Lambda_U$, Fa\`a-di-Bruno with sup-jets
$L_{\le6}$, times $e_0$; the inverse rides $|\det\Psi(G)|$ bounded below by
(u1); $\omega=e^{-1}\partial e+e^{-1}\Gamma e$ in $C^{5,\alpha}$ with
$\|\Gamma[g]\|_{C^{5,\alpha}}\le c\Lambda_U^9d_U^{-6}$ (port L1 at $k=5$).
\eqref{dspin:eq-framediff}: $e[g]-e[g']=(\int_0^1 D\Psi(G_\tau)\,d\tau)[G-G']
\,e_0$ along the real segment $G_\tau=(1-\tau)G'+\tau G$ (interior to
$B_{\rho_\eta/2}$), $G-G'=(e_0^{-1})^{\mathsf T}h\,e_0^{-1}$ --- \emph{the M3
mechanism exactly}, a fixed analytic map differenced along the real segment,
not a variation in $g$; the connection Leibniz telescope contributes
$\Delta(e^{-1})$, $\partial(\Delta e)$, and $\Delta\Gamma=\delta\Gamma$
($\kappa_\Gamma(5)$, port L1).
\end{proof}

\paragraph{The spinor parametrix and its normally-hyperbolic reduction.}
The massive Dirac operator $\Dsl=\gamma^\mu\nabla_\mu$ on $S$ obeys the
Lichnerowicz--Weitzenb\"ock identity
\begin{equation}\label{dspin:eq-lich}
  P:=\Dsl^2\mp m^2=\square^\nabla+E,\qquad
  E:=(\tfrac14 R\mp m^2)\,\mathbf 1_S,
\end{equation}
with $\square^\nabla=g^{\mu\nu}\nabla_\mu\nabla_\nu$ the spin-connection wave
operator (principal part $=\square_g$ \emph{exactly}) and $E$ the
\emph{central} (scalar-curvature $\times$ identity) endomorphism --- by the
first Bianchi identity there is no Ricci/Weyl part on $S$; the sign of the
mass term is a signature/convention choice that enters only a tracked constant.
$P$ is normally hyperbolic, so the bundle Hadamard/DeWitt parametrix exists
per metric (Moretti~\cite[Thm.~2.1]{Moretti2003}, stated for
normally-hyperbolic \emph{bundle} operators --- the same theorem the scalar
port's Lem.~\ref{port:lem-vk} cites; classical origins
Dimock (1982), DeWitt--Brehme, B\"ar--Ginoux--Pf\"affle; fixed-$g$
Dirac Hadamard renormalisation in print,
\cite{DappiaggiHackPinamonti2009}). \emph{Only the family-uniform
finite-regularity form is new here}, mirroring the scalar port. Write
\begin{equation}\label{dspin:eq-Hdef}
  H^{\mathrm{sp}}_g=\sum_{a\le N(K)}\theta_a\,Z^{(2),a}_{\mathrm{sp},g},\quad
  Z^{(2)}_{\mathrm{sp},g}=\frac1{8\pi^2}\Bigl[\frac{\Delta^{1/2}_g\,I_g}
  {\sigma_{g,\epsilon}}+V^{(2)}_g\log\frac{\sigma_{g,\epsilon}}{\ell^2}\Bigr],
  \quad V^{(2)}_g=\sum_{k=0}^2 V_k[g]\,\sigma_g^k,
\end{equation}
with the same $g$-independent cutoffs $\theta_a$, the same single length
$\ell$, the same $N=2$ truncation and $\sigma_{g,\epsilon}$-prescription as
\eqref{port:eq-Hdef}; $I_g(x,y)\in\mathrm{End}(S)$ is the bispinor parallel
propagator ($\nabla_{\nabla\sigma}I_g=0$, $I_g(x,x)=\mathbf 1_S$) and
$V_k[g]\in\mathrm{End}(S)$ the bundle Hadamard coefficients. Both $\sigma_g$
and $\Delta^{1/2}_g$ are the \emph{scalar} metric objects.

\begin{lemma}[bundle Hadamard coefficients: ball-uniform sup and difference;
the register is unchanged]\label{dspin:lem-rec}
Let $I_g$ solve, in the rescaled normal form of Lem.~\ref{port:lem-sigma}(i),
$(d/dr)I+\Omega(x,\omega,r)I=0$, $\Omega:=\omega[g](\dot\xi)\in\mathrm{End}(S)$,
$I(x,x)=\mathbf 1$; and $V_k$ the transport recursion
\begin{equation}\label{dspin:eq-rec}
  2\nabla_{\nabla\sigma}V_k+(\square\sigma-2n+4k)V_k=-(\square^\nabla+E)V_{k-1},
  \qquad V_{-1}:=0,\quad V_0=\Delta^{1/2}_g I_g.
\end{equation}
Then, on the collar $V_{\delta_c}$ (fiber operator norm), with
$\Lambda_R:=\sup_U\|R[g]\|_{C^{0,\alpha}(V)}$,
\begin{align}
  \|I_g\|_{C^{1,\alpha}(V)}&\le\Lambda_I,& \|I_g-I_{g'}\|_{C^{0,\alpha}(V)}
  &\le\kappa_I\,\|h\|_{C^{1,\alpha}(K'')},\label{dspin:eq-Ibnd}\\
  \|V_k[g]\|_{C^{0,\alpha}(V)}&\le\Lambda_V,& \|V_k[g]-V_k[g']\|_{C^0(V)}
  &\le\kappa_V\,\|h\|_{C^{2k+2}(K'')}\le\kappa_V\|h\|_{C^6},\label{dspin:eq-Vbnd}
\end{align}
with $\Lambda_I=c\,e^{c\delta_c\Lambda_\omega}(1+\Lambda_S)^2$,
$\kappa_I=c\Lambda_I^2(\kappa_\omega(0)+\Lambda_\omega\kappa_{\mathrm{fl}})$,
$E[g]=(\tfrac14R\mp m^2)\mathbf 1_S$ with $\|E_g-E_{g'}\|_{C^0}\le\kappa_E
\|h\|_{C^2}$ ($\kappa_E\le c\Lambda_U^3 d_U^{-2}$), and
\[
  \Lambda_V=c\,e^{c(\Lambda_S+\Lambda_\omega)\delta_c}\,
  \mathsf P(\Lambda_S,\Lambda_\Delta,\Lambda_I,\Lambda_\omega,\Lambda_R,
  \Lambda_U,d_U^{-1}),\quad
  \kappa_V=c\,e^{c(\Lambda_S+\Lambda_\omega)\delta_c}\,\mathsf P\,
  (1+\kappa_S+\kappa_\Gamma(1)+\kappa_\omega(1)+\kappa_E).
\]
All are ball-uniform, state-independent. \emph{Register verdict:} the deepest
metric jet consumed, linearly, is $\partial^6 g\in C^{0,\alpha}$ at $V_2$ ---
\emph{identical} to the scalar $v_2$ (Lem.~\ref{port:lem-vk} proof); the fence
stays $s>n/2+6=8$, with no lift.
\end{lemma}
\begin{proof}
$I_g$: a linear transport ODE with $C^3$-flow coefficient $\omega[g]$
(Lem.~\ref{dspin:lem-frame}); Gronwall on the length-$\le\delta_c$ collar
geodesic gives $\Lambda_I$, its variational (inverse-flow) difference gives
$\kappa_I$ --- exactly Lem.~\ref{port:lem-sigma}(ii)'s Gronwall-along-geodesics
with $\omega$ in place of $\Gamma$, \emph{not} a new estimate class. In
\eqref{dspin:eq-rec} the transport correction $[\omega(\nabla\sigma),\cdot]$ is
$O(r)$ ($\nabla\sigma=O(r)$ at coincidence), folded into the ``$+O(r)$'' of the
indicial coefficient $A_k=\square\sigma-2n+4k+O(r)$; the indicial root
$A_k(0)=4k>0$ ($k\ge1$; $k=0$ fixed by the $V_0$ anchor) is \emph{unchanged},
so the regular-singular Duhamel/Gronwall of Lem.~\ref{port:lem-vk} transcribes
with the same rates, the unknown now $\mathrm{End}(S)$-valued (matrix Gronwall
in the fiber norm; the fiber constants of $n$ absorb into $c_n$). Jet count:
the source $-(\square^\nabla+E)V_{k-1}$ has top metric jet
$\partial^2 V_{k-1}\!\supset\partial^{2k+2}g$ (the leg the scalar $\square$
also has); every extra term is a product with a strictly shallower top factor
--- $\omega\,\partial V_{k-1}\!\to\!\partial^{2k+1}g$,
$(\partial\omega)V_{k-1},\,\omega^2V_{k-1},\,EV_{k-1}\!\to\!\partial^{2k}g$ ---
because $E=\tfrac14R\mp m^2$ and $\partial\omega$ are clean metric $2$-jets
(the frame is \emph{algebraic} in $g$: Lem.~\ref{dspin:lem-frame}). Hence the
ledger equals the scalar one line for line ($\Delta^{1/2}I:\|h\|_{C^4}$,
$V_0:\|h\|_{C^2}$, $V_1:\|h\|_{C^4}$, $V_2:\|h\|_{C^6}$); the $C^{0,\alpha}$
grade rides one order down of the $C^{6,\alpha}$ metric and $C^{5,\alpha}$
frame data (frame margin $\ge2$). The $\mp m^2$ constant enters $\Lambda_V$
additively; the bundle rank contributes the fixed factor $N_{\mathrm{sp}}$.
\end{proof}

\begin{lemma}[spinor clause (i), difference form, over the full $H^s$ ball;
every constant displayed]\label{lem:dirac-parametrix-ball}
With $U,s,\alpha$ as above, there is a ball-uniform, state-independent
constant $C_H^{\mathrm{sp}}(K)=C_{\mathrm{sob}}(K)^3\,\kappa^{\mathrm{sp}}
(g_0,K)$ such that for all $g,g'\in U$ and all $u,v$ supported in $K$,
\begin{equation}\label{dspin:eq-clausei}
  \bigl|(H^{\mathrm{sp}}_g-H^{\mathrm{sp}}_{g'})(u\otimes v)\bigr|
  \le C_H^{\mathrm{sp}}(K)\,\|g-g'\|_{H^s}\,\|u\|_{H^{s-2}}\|v\|_{H^{s-2}},
\end{equation}
$u,v$ now sections of $S$, $H^{s-2}=H^{s-2}(K;S)$. The same $C_{\mathrm{sob}}$
and cube as \eqref{port:eq-csob} (frame and connection differences ride the
\emph{same} $C^{6,\alpha}\hookleftarrow H^s$ embedding as $\|h\|_{C^6}$), and
\[
  \kappa^{\mathrm{sp}}=\frac{|K|\,m(K)\,c_\theta\,\kappa_P^{\mathrm{sp}}}{8\pi^2}
  \bigl[\underbrace{\kappa_{\mathrm{lead}}\tfrac{\delta_c^2}2}_{(A^{\mathrm{sp}})}
  +\underbrace{c\Lambda_I\Lambda_\Delta\kappa_S\tfrac{\delta_c^2}2}
  _{(B^{\mathrm{sp}})}
  +\underbrace{\kappa_C'^{\mathrm{sp}}\delta_c^2(1+|\log\tfrac{\delta_c^2}
  {\ell^2}|)}_{(C^{\mathrm{sp}})}
  +\underbrace{c\Lambda_V(1+\Lambda_\sigma\delta_c^2)^2\kappa_S
  \tfrac{\delta_c^4}4}_{(D^{\mathrm{sp}})}\bigr],
\]
$\kappa_P^{\mathrm{sp}}=N_{\mathrm{sp}}\kappa_P$ (Lem.~\ref{port:lem-pair}, the
End-valued amplitude a passive spectator up to the trace contraction),
$\kappa_{\mathrm{lead}}=\Lambda_I\kappa_\Delta+\Lambda_\Delta\kappa_I$, and
$\kappa_C'^{\mathrm{sp}}=c[\kappa_V(1+\Lambda_\sigma\delta_c^2)^2+\Lambda_V
\kappa_\sigma\delta_c^2(1+\Lambda_\sigma\delta_c^2)]$. It holds a fortiori over
$U_\omega\times U_\omega$.
\end{lemma}
\begin{proof}
The leading term differences by the exact calculus identities of
\eqref{port:eq-clausei}'s proof; the sole new split is
$\Delta^{1/2}_g I_g-\Delta^{1/2}_{g'}I_{g'}

$|\int\bar u\,\Phi\,\sigma_\epsilon^{-p}v|\le\sup\|\Phi\|_{\mathrm{op}}\cdot

\begin{remark}[massless case]

\end{remark}
